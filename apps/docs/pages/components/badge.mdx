import { Badge, HStack, Text, VStack } from "@superstyling/core";
import { ComponentDemo } from "../../components/ComponentDemo";
import { PropsTable } from "../../components/PropsTable";

export const COLOR_SCHEMES = ["gray", "blue", "red", "green", "purple"];
export const VARIANTS = ["solid", "subtle", "outline"];

# Badge

Compact status or count indicator. Supports three visual variants and the full `colorScheme` palette.

## Import

```tsx
import { Badge } from "@superstyling/core";
```

## Variants × colorScheme

<ComponentDemo
  code={`{VARIANTS.map((variant) => (
  <HStack key={variant} gap="$2">
    {COLOR_SCHEMES.map((cs) => (
      <Badge key={cs} variant={variant} colorScheme={cs}>{cs}</Badge>
    ))}
  </HStack>
))}`}
  preview={
    <VStack gap="$3">
      {VARIANTS.map((variant) => (
        <HStack key={variant} gap="$2" alignItems="center">
          <Text width={70} fontSize="$2" fontWeight="600">{variant}</Text>
          {COLOR_SCHEMES.map((cs) => (
            <Badge key={cs} variant={variant} colorScheme={cs}>{cs}</Badge>
          ))}
        </HStack>
      ))}
    </VStack>
  }
/>
