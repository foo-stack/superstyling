import { Alert, VStack } from "@superstyling/core";
import { ComponentDemo } from "../../components/ComponentDemo";
import { PropsTable } from "../../components/PropsTable";

export const STATUSES = ["info", "success", "warning", "error", "loading"];
export const VARIANTS = ["subtle", "solid", "left-accent", "top-accent"];

export function titleCase(s) {
  return s.charAt(0).toUpperCase() + s.slice(1);
}

# Alert

Callout for communicating status. Compound shape: `Alert.Icon`, `Alert.Content` with `Alert.Title` + `Alert.Description` inside. Five statuses × four visual variants.

## Import

```tsx
import { Alert } from "@superstyling/core";
```

## Basic usage

<ComponentDemo
  code={`<Alert status="success">
  <Alert.Icon />
  <Alert.Content>
    <Alert.Title>Saved</Alert.Title>
    <Alert.Description>Your changes have been persisted.</Alert.Description>
  </Alert.Content>
</Alert>`}
  preview={
    <Alert status="success">
      <Alert.Icon />
      <Alert.Content>
        <Alert.Title>Saved</Alert.Title>
        <Alert.Description>Your changes have been persisted.</Alert.Description>
      </Alert.Content>
    </Alert>
  }
/>
