# Tabs

<StatusRow tier="1" platforms="Cross-platform" since="v0.2.0" />

Switch between multiple panels with keyboard-navigable tabs. Built on `@tamagui/tabs` — WAI-ARIA tablist / tab / tabpanel semantics with roving tabindex + ←/→/Home/End keyboard nav.

## Import

```tsx
import { Tabs } from "@superstyling/core";
```

## Basic

<ComponentDemo
  code={`<Tabs defaultValue="profile">
  <Tabs.List>
    <Tabs.Tab value="profile">Profile</Tabs.Tab>
    <Tabs.Tab value="settings">Settings</Tabs.Tab>
    <Tabs.Tab value="billing">Billing</Tabs.Tab>
  </Tabs.List>
  <Tabs.Panels>
    <Tabs.Panel value="profile">Profile content</Tabs.Panel>
    <Tabs.Panel value="settings">Settings content</Tabs.Panel>
    <Tabs.Panel value="billing">Billing content</Tabs.Panel>

</Tabs>`}
  preview={

        <Tabs.Tab value="profile">Profile</Tabs.Tab>

      </Tabs.List>
